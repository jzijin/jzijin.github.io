最近点对问题

考虑确定平面上n个点(x1, y1),(x2,y2)…(xn,yn)集合上的最近点对的问题。

可以很简单的写出暴力法破解：
// Points 为所有点对的集合，using Point = pair<double, double>, 
// a, b 传出参数， 最近点对的两个点。
double get_distance_enforce(const vector<Point> &Points, Point &a, Point &b)
{
    double distance = static_cast<double>(INT32_MAX);
    for (auto i = Points.begin(); i != Points.end(); ++i)
    {
        for (auto j = i + 1; j != Points.end(); ++j)
        {
            double tmp = sqrt(distance_square(*i, *j));
            // 如果跨分割线的两点距离小于已知距离，则记录该距离。
            if (tmp < distance)
            {
                distance = tmp;
                a = *i;
                b = *j;
            }
        }
    }
    return distance;
}
这样得出的算法复杂度为O(n^2).

方法2：可以用递归的方法，分而治之，将一个大问题分解为若干个小问题。

#include <iostream>
#include <random>
#include <chrono>
#include <cmath>
#include <algorithm>
using namespace std;
using Point = pair<double, double>;
constexpr int NO_DISTANCE = 10000;

// 随机生成点对
void set_point(vector<Point> &Points, int size, double x, double y)
{
    std::default_random_engine e(std::chrono::system_clock().now().time_since_epoch().count());
    std::uniform_real_distribution<double> dis(x, y);
    for (auto i = 0; i < size; ++i)
    {
        Points.push_back(Point(dis(e), dis(e)));
    }
}

// 打印点对
void print_points(const vector<Point> &Points)
{
    for (auto i : Points)
    {
        cout << "(" << i.first << ", " << i.second << ")" << endl;
    }
    cout << endl;
}

double distance_square(Point a, Point b)
{
    return (a.first - b.first) * (a.first - b.first) + (a.second - b.second) * (a.second - b.second);
}

// points 所有的点的集合， a, b 传出参数
// Points_X 按X排序的序列， Points_Y 按Y排序的序列。
double closestPoint(const vector<Point> &Points_X, const vector<Point> &Points_Y, Point &a, Point &b)
{
    Point a1, b1, a2, b2;
    int length = Points_X.size();
    double distance;
    if (length < 2)
        return NO_DISTANCE;
    else if (length == 2)
    {
        a = Points_X[0];
        b = Points_X[1];
        distance = sqrt(distance_square(Points_X[0], Points_X[1]));
    }
    else
    {
        vector<Point> pst1(Points_X.begin(), Points_X.begin() + length / 2);
        vector<Point> pst2(Points_X.begin() + length / 2, Points_X.end());
        double d1 = closestPoint(pst1, Points_Y, a1, b1);
        double d2 = closestPoint(pst2, Points_Y, a2, b2);
        if (d1 < d2)
        {
            distance = d1;
            a = a1;
            b = b1;
        }
        else
        {
            distance = d2;
            a = a2;
            b = b2;
        }

        // 一个点在左边 一个点在右边的情况。
        vector<Point> pst3;
        double mid = (Points_X.begin() + length / 2 - 1)->first;
        for (auto point : Points_Y)
        {
            if (abs(point.first - mid) <= distance)
                pst3.push_back(point);
        }

        // 只需要与有序的链接的7个点进行比较就可以了。
        for (auto i = pst3.begin(); i != pst3.end(); ++i)
        {
            for (auto j = i + 1; j != i + 7 && j != pst3.end(); ++j)
            {
                double tmp = sqrt(distance_square(*i, *j));
                // 如果跨分割线的两点距离小于已知距离，则记录该距离。
                if (tmp < distance)
                {
                    distance = tmp;
                    a = *i;
                    b = *j;
                }
            }
        }
    }
    return distance;
}

数学归纳法和强归纳法 

数学归纳法的原理： 为证明对所有的正整数n，p(n)为真，其中p(n)是一个命题函数
需要完成两个步骤
    基础步骤： 证明命题p(1)成立
    归纳步骤： 证明对每个正整数k来说， p(k) --> p(k+1)

    强归纳法： 要证明对所有的正整数n而言，都有p(n)为真，其中p(n)为命题函数，我们
要完成如下两个步骤：
    基础步骤： 证明p(1)为真
    归纳步骤： 要证明对所有正整数k来说，蕴含式 p(1)&p(2)...p(k)  --> p(k+1)也为真



example：
a) Determine which amounts of postage can be formed
using just 4-cent and 11-cent stamps.
b) Prove your answer to (a) using the principle of math-
ematical induction. Be sure to state explicitly your
inductive hypothesis in the inductive step.
c) Prove your answer to (a) using strong induction. How
does the inductive hypothesis in this proof differ from
that in the inductive hypothesis for a proof using math-
ematical induction?

a)、4,8,11,12,15,16,19,20,22,23,24,26,27,28，以及大于等于30的所有值
b)、令P(n)是命题：我们可以只用4分邮票和11分邮票组成n分邮资。证明：对于所有 n >= 30, P(n)为真。
基础步骤：可以组成k分邮资，如何组成k+1分邮资
归纳步骤：如果k分邮资中有一张11分邮票，可以用3张4分邮资代替组成k+1；否则k分邮资只由4分邮票构成。因为k>=30,可以用三张11分邮票代替8张4分邮票，组成k+1分邮资。
c)P(n)与b中相同。
基础步骤：P(30) = 11+11+4, p(31) = 11+4+4+4+4+4, p(32)=4+4+4+4+4+4+4+4, p(33) = 11+11+11;
归纳步骤：假设归纳假设p(j)对于所有j(30<=j<=k)都为真，这里k是大于等于33的任意整数。我们要证明p(k+1)为真。因为k-3>=30,所以p(k-3)为真，即可以组成k-3分邮资。在这个信封上在贴一张4分邮票，就组成了k+1分邮资。在这个证明中，我们的归纳假设是p(j)对所有介于30和k之间的值都为真，而不只是p(30)为真。

#include <iostream>
#include <vector>
using namespace std;

// 找出列表中的第一只单身狗
int find_single_dog(const vector<int> &match_for_man)
{
    for(auto i=0; i<match_for_man.size(); ++i)
    {
        if(match_for_man[i] == -1)
            return i;
    }
    return -1;
}

// 比较自己和原配的排名 看看能否成功挖墙脚
bool check_rank(const vector<int> &preference_of_women, int man_index, int old_man_index)
{
    for(auto i:preference_of_women)
    {
        if(i == man_index)
            return true;
        else if(i == old_man_index)
            return false;
    }
    return false;

}

// 打印结果
void print_res(const vector<int>& match_for_man, const vector<int>& match_for_woman)
{
    for(auto i=0; i<match_for_man.size(); ++i)
    {
        cout << "男生" << i << "号 --->" << " 女生" << match_for_man[i] << "号" << endl;
    }
    for(auto i=0; i<match_for_woman.size(); ++i)
    {
        cout << "女生" << i << "号 --->" << " 男生" << match_for_woman[i] << "号" << endl;
    }
}
int main()
{
    // 男生的喜好列表
    vector<vector<int>> preference_of_man{vector<int>{0, 1, 3, 2}, vector<int>{0, 2, 3, 1}, vector<int>{3, 0, 2, 1}, vector<int>{0, 3, 1, 2}};
    // 女生的喜好列表
    vector<vector<int>> preference_of_woman{vector<int>{3, 0, 1, 2}, vector<int>{3, 2, 1, 0}, vector<int>{3, 2, 0, 1}, vector<int>{2, 3, 0, 1}};
    int length = preference_of_man.size();
    vector<int> match_for_man(length, -1);
    vector<int> match_for_woman(length, -1);
    // vector<int> matched_len(length, -1);
    int man_index;
    int woman_index;
    // 循环查找列表中的单身狗
    while((man_index = find_single_dog(match_for_man)) > -1)
    {
        print_vec(match_for_man);
        for(int index=0; index < length; ++index)
        {
            // 找出该单身狗所心仪的对象
            if(match_for_woman[woman_index] == -1)
            {
                match_for_man[man_index] = woman_index;
                match_for_woman[woman_index] = man_index;
                break;
            }
            // 很不幸 该对象已经有伴了。。。 看一下能不能挖走墙角
            else if(check_rank(preference_of_woman[woman_index], man_index, match_for_woman[woman_index]))
            {
                // 如果能挖走墙角
                // 踢掉原配
                match_for_man[match_for_woman[woman_index]] = -1;
                match_for_woman[woman_index] = man_index;
                match_for_man[man_index] = woman_index;
                break;
            }
        }
    }

    print_res(match_for_man, match_for_woman);
    return 0;
}

背包问题 

int knapsack(int weight, const vector<int>& things)
{
    vector<int> dp(weight+1);
    for(auto i:things)
    {
        // 初始化背包数组
        dp[i] = i;
    }
    for(auto i=1; i<=weight; ++i)
    {
        // 跳过哪些已经被赋值的元素
        if(dp[i])
            continue;
        vector<int> tmp;
        for(auto thing:things) 
        {
            if(i-thing > 0)
                // 添加可能的值进入dp数组
                tmp.push_back(dp[i-thing] + thing);
        }
        if(tmp.size())
            // 递推计算可能的最大的值
            dp[i] = *max_element(tmp.cbegin(), tmp.cend());
    }
    return dp[weight];

}

    }
}

void BreadthFirstSearch(BitNode *root)
{
    queue<BitNode*> nodeQueue;

    {
        BitNode *node = nodeQueue.front();
        cout << node->data << ' ';
        nodeQueue.pop();
        if (node->left)
        {
            nodeQueue.push(node->left);

        {
            nodeQueue.push(node->right);
        }
    }

chapter 1 了解MySQL
1.1 数据库基础
1.1.1 什么是数据库
    数据库(database)：保存有组织的数据的容器（通常是一个文件或一组文件）。
1.1.2 表
    表(table) 某种特定类型数据的结构化清单
    模式(schema) 关于数据库和表的布局及特性的信息
1.1.3 列和数据类型
    列(column) 表中的一个字段。所有表都是由一个或多个列组成的
    数据类型(datatype) 所容许的数据的类型。每个表列都有相应的数据类型，
    它限制（或容许）该列中存储的数据
1.1.4 行
    行(row) 表中的一个记录
1.1.5 主键
    主键(primary key) 一列（或一组列），其值能够唯一区分表中每个行

        任何两行都不具有相同的主键值；
        每个行都必须具有一个主键值

    主键的最好习惯：

// 选择排序
/*
给定数组：int[] arr={里面n个数据}；第1趟排序，在待排序数据arr[1]~arr[n]中选出最小的数据，将它与arrr[1]交换；第2趟，在待排序数据arr[2]~arr[n]中选出最小的数据，将它与r[2]交换；以此类推，第i趟在待排序数据arr[i]~arr[n]中选出最小的数据，将它与r[i]交换，直到全部排序完成。
*/
void selectsort(int *arr)
{
    for(int i=0; i<size-1; i++)
    {
        int k=i;
        for (int j=k+1; j<size; j++)
        {
            if(arr[j] < arr[k])
            {
                k = j; // 记录目前找到的最小元素的位置
            }
        }
        // 内层循环后，也就是找到本轮循环的最小的数之后，在进行交换
        if(i != k)
        {
            int temp = arr[i];
            arr[i] = arr[k];
            arr[k] = temp;
        }
    }
}
// 快速排序
void quicksort(int left, int right)
{
    int i, j, t, temp;
    if (left > right)
    {
        return;
    }
    temp = a[left]; // temp 中存的就是基准数
    i = left;
    j = right;
    while (i != j)
    {
        // 顺序很重要， 要先从右往左找
        while (a[j] >= temp && i < j)
        {
            j--;
        }
        while (a[i] <= temp && i < j)
        {
            i++;
        }
        // 交换两个数在数组中的位置
        if (i < j)
        { // 当哨兵i和哨兵j没有相遇时
            t = a[i];
            a[i] = a[j];
            a[j] = t;
        }
    }
    // 最终基准数归位

    a[left] = a[i];
    a[i] = temp;
    quicksort(left, i - 1);  // 继续处理左边的 这是递归的过程
    quicksort(i + 1, right); // 处理右边的， 这也是一个递归的过程
}

*/

/* 当数比较集中的时候非常适合用桶排序 可以极大的节省时间
一年的全国高考考生人数为500 万，分数使用标准分，最低100 ，最高900 ，没有小数，要求对这500 万元素的数组进行排序。
分析：对500W数据排序，如果基于比较的先进排序，平均比较次数为O(5000000*log5000000)≈1.112亿。但是我们发现，这些数据都有特殊的条件： 100=<score<=900。那么我们就可以考虑桶排序这样一个“投机取巧”的办法、让其在毫秒级别就完成500万排序。
方法：创建801(900-100)个桶。将每个考生的分数丢进f(score)=score-100的桶中。这个过程从头到尾遍历一遍数据只需要500W次。然后根据桶号大小依次将桶中数值输出，即可以得到一个有序的序列。而且可以很容易的得到100分有***人，501分有***人。
实际上，桶排序对数据的条件有特殊要求，如果上面的分数不是从100-900，而是从0-2亿，那么分配2亿个桶显然是不可能的。所以桶排序有其局限性，适合元素值集合并不大的情况。

        for (i = 0; i < n - 1 - j; i++)// 内层循环控制每一趟排序多少次
        {
            if (a[i] > a[i + 1])
            {
                temp = a[i];
                a[i] = a[i + 1];
                a[i + 1] = temp;
            }
        }
}

def merge_pass(lfrom, lto, llen, slen):
    i = 0
    while i+2*slen < llen:    # 归并长slen的两段

    if i+slen < llen   # 剩下两段， 后段长度小于slen

def merge_sort(lst):

        slen *= 2

void Merge(int sourceArr[], int tempArr[], int startIndex, int midIndex, int endIndex)
{
    int i = startIndex, j = midIndex + 1, k = startIndex;
    while (i != midIndex + 1 && j != endIndex + 1) // 从小到大记录数据
    {
        if (sourceArr[i] > sourceArr[j])
            tempArr[k++] = sourceArr[j++];
        else
            tempArr[k++] = sourceArr[i++];
    }
    while (i != midIndex + 1) // 记录前面一段的剩余部分
        tempArr[k++] = sourceArr[i++];
    while (j != endIndex + 1) // 记录后面一段的剩余部分
        tempArr[k++] = sourceArr[j++];
    for (i = startIndex; i <= endIndex; i++)
        sourceArr[i] = tempArr[i]; // 重新写回
}

void MergeSort(int sourceArr[], int tempArr[], int startIndex, int endIndex)
{

        MergeSort(sourceArr, tempArr, startIndex, midIndex);
        MergeSort(sourceArr, tempArr, midIndex + 1, endIndex);
        Merge(sourceArr, tempArr, startIndex, midIndex, endIndex);
    }
}

/*
A[i]的左节点为A[2i+1],右节点为A[2i+2]，父节点为A[i/2]，它从数组索引的角度描述了数字与数字在二叉树中的位置关系。

*/

/*堆序性说得通俗一点儿就是，父节点中的元素不小于（不大于）任意子节点的元素。
这里的元素在本文中是以数字体现的。注：本文只讨论“大根堆”，即父节点中的元素不小于任意子节点的元素这种情形。
所以，在一个大根堆中，一个节点的元素在其子树所有元素组成的集合中必定是最大值。这一结论至关重要。
以下给出了两个大根堆的示例：
*/
/*
下面开始进入正题了，堆排序算法究竟是怎么实现的呢？其实只需要简单的两步就可以实现堆排序：1）根据无序数组创建一个大根堆；2）不断调整大根堆，使其到达有序。
*/

#include <stdio.h>
#include <stdlib.h>

void swap(int* a, int* b) {
    int temp = *b;
    *b = *a;
    *a = temp;
}

void max_heapify(int arr[], int start, int end) {
    //建立父节点指标和子节点指标
    int dad = start;
    int son = dad * 2 + 1;
    while (son <= end) { //若子节点指标在范围内才做比较
        if (son + 1 <= end && arr[son] < arr[son + 1]) //先比较两个子节点大小，选择最大的
            son++;
        if (arr[dad] > arr[son]) //如果父节点大於子节点代表调整完毕，直接跳出函数
            return;
        else { //否则交换父子内容再继续子节点和孙节点比较
            swap(&arr[dad], &arr[son]);
            dad = son;
            son = dad * 2 + 1;
        }
    }
}

void heap_sort(int arr[], int len) {
    int i;
    //初始化，i从最後一个父节点开始调整
    for (i = len / 2 - 1; i >= 0; i--)
        max_heapify(arr, i, len - 1);
    //先将第一个元素和已排好元素前一位做交换，再重新调整，直到排序完毕
    for (i = len - 1; i > 0; i--) {
        swap(&arr[0], &arr[i]);
        max_heapify(arr, 0, i - 1);
    }
}

int main() {
    int arr[] = { 3, 5, 3, 0, 8, 6, 1, 5, 8, 6, 2, 4, 9, 4, 7, 0, 1, 8, 9, 7, 3, 1, 2, 5, 9, 7, 4, 0, 2, 6 };
    int len = (int) sizeof(arr) / sizeof(*arr);
    heap_sort(arr, len);
    int i;
    for (i = 0; i < len; i++)
        printf("%d ", arr[i]);
    printf("\n");
    return 0;
}

// 剃须刀排序 双向冒泡排序
vector<int> shaker_sort(const vector<int> &v)
{
    vector<int> res = v;
    int start = 0, end = res.size() - 1, shift = 1;
    while(start < end)
    {
        for(auto i = start; i<end; ++i)
        {
            if(res[i] > res[i+1])
            {
                swap(res[i], res[i+1]);
                shift = i;
            }
        }
        // 到shift 以后的元素已经排好序了
        end = shift;
        for(auto j=end-1; j>=start; --j)
        {
            if(res[j] > res[j+1])
            {
                swap(res[j], res[j+1]);
                shift = j + 1;
            }
        }
        // 到shift之前的元素已经都排好序了。
        start = shift;
    }
    return res;
}

void PostOrder(struct TreeNode *root)
{
    s = STACK()
    while(root != NULL || ！s.is_empty())
    {
        while(root != NULL) // 直到栈顶的两子树为空
        {
            s.push(root)
            root = root->left if root->left != NULL else root->right // 能左则左 否则向右

        proc(root->val);// 处理数据
