背包问题 

int knapsack(int weight, const vector<int>& things)
{
    vector<int> dp(weight+1);
    for(auto i:things)
    {
        // 初始化背包数组
        dp[i] = i;
    }
    for(auto i=1; i<=weight; ++i)
    {
        // 跳过哪些已经被赋值的元素
        if(dp[i])
            continue;
        vector<int> tmp;
        for(auto thing:things) 
        {
            if(i-thing > 0)
                // 添加可能的值进入dp数组
                tmp.push_back(dp[i-thing] + thing);
        }
        if(tmp.size())
            // 递推计算可能的最大的值
            dp[i] = *max_element(tmp.cbegin(), tmp.cend());
    }
    return dp[weight];

}
